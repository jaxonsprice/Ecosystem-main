_/**
 * Flock class

class Flock{
    /**
     * constructor
     */
    constructor() {
        this.boids = []
    }
    /**
     * runs the boids
     */
    run() {
        for (let boid of this.boids) {
            boid.run(this.boids)
        }
    }
    /**
     * adds a boid to the array
     * @param {Object} boid 
     */
    addBoid(boid) {
        this.boids.push(boid)
    }
}
